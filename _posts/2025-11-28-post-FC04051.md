import turtle
import random
def tree(l,a,s):
   if l>30: 
    turtle.pensize(s)
    turtle.forward(l)
    turtle.left(a)
    tree(l*0.7,a,s-1)
    turtle.right(2*a)
    tree(l*0.7,a,s-1)
    #turtle.color("green")
    turtle.left(a)
    turtle.backward(l)
   else:
      turtle.color("black")
      turtle.begin_fill()
      turtle.circle(5)   
      turtle.end_fill() 
   
   

turtle.setheading(90)
def back():
   
   for i in range(4):
      turtle.forward(2000)
      turtle.right(90)
   turtle.fillcolor("dark orange")
      
def zamin():
   for i in range(4):
      turtle.forward(1400)
      turtle.right(90)
   turtle.fillcolor("black")
            

def makekh():
   m=turtle.Turtle()
   m.shape("circle")
   m.shapesize(6)
   m.fillcolor("red")
   m.penup()
   m.setpos(-650,300)
   

def home1():
   for i in range(4):
      turtle.forward(120)
      turtle.right(90)
   turtle.fillcolor("black")

def home2():

   t.shape("triangle")  
   t.shapesize(6.5)

   t.setpos(-490,48)
   t.fillcolor("black")
   t.setheading(90)

def water(x,y):
   b=turtle.Turtle()
   b.shape("square")
   b.shapesize(stretch_len=0.06,stretch_wid=0.55)
   b.color("black")
   b.penup()
   b.setpos(x,y)
   b.pendown()
   b.setheading(-35)
      

turtle.tracer(0)
def rain(n):
   for i in range(n):
      x=random.randint(-650,600)
      y=random.randint(-85,600)
      water(x,y)
   turtle.update()   

def J(n):
  for i in range(n):
        
        l=random.randint(130,180)
        s=random.randint(5,6)
        a=random.randint(28,35)
        x=random.randint(-50,570)
        y=random.randint(-150,-90)
        turtle.penup()
        turtle.setpos(x,y) 
        turtle.pendown()
        tree(l,a,s)
